I2C Example
I2C Driver

This example demonstrates typical use of the Inter-Integrated Circuits (I2C) Driver.

Source code:
/******************************************************************************
* (c) Copyright 2010-2013, Freescale Semiconductor Inc.
* ALL RIGHTS RESERVED.
******************************************************************************
* i2c_test.c
******************************************************************************/
#include "drivers.h"
uint8 ui8_data[4] = {0,0,0,0};
void main (void)
{
/* initialize SIM module */
SIM_Init (SIM_MODULE_ALL_PERIPH_ON_CONFIG);
/* clock mode 2:1:1, 48MHz */
SIM_SetClkMode (SYSCLK_MODE1);
SIM_SetClkDiv (SYSCLK_DIV1);
FLL_Init (FLL_MODULE_FEE_48MHZ_CONFIG);
/* PORTs initialize, PTD7_SCL, PTE0_SDA */
PORT_Init (PORTD, PORT_MODULE_ALT2_MODE, PIN7);
PORT_Init (PORTE, PORT_MODULE_ALT2_MODE, PIN0);
/* I2C module initialize */
I2C_Init (I2C0, I2C_MODULE_100K_MASTER_IRQ_DI_CONFIG);
/* Read 4 data bytes for EEPROM CAT24C05 */
/* Set TX mode and Generate START on I2C bus */
I2C_SetTxMode(I2C0);
I2C_StartSignal(I2C0);
/* Send data to Slave, Wait on action complete, Clear Irq. flag */
I2C_WrData(I2C0, 0xAC);
while(I2C_GetIsrFlag(I2C0) == 0);
I2C_ClrIsrFlag(I2C0);
/* Send data to Slave, Wait on action complete, Clear Irq. flag */
I2C_WrData(I2C0, 0x00);
while(I2C_GetIsrFlag(I2C0) == 0);
I2C_ClrIsrFlag(I2C0);
/* Generate Repeat START on I2C bus */
I2C_RepeatStartSignal(I2C0);
/* Send data to Slave, Wait on action complete, Clear Irq. flag */
I2C_WrData(I2C0, 0xAD);
while(I2C_GetIsrFlag(I2C0)== 0);
I2C_ClrIsrFlag(I2C0);
/* Set to RX Mode */
I2C_SetRxMode(I2C0);
ui8_data[0] = I2C_RdData(I2C0);
/* Wait on action complete, Read data_0, Clear Irq. flag */
while(I2C_GetIsrFlag(I2C0) == 0);
ui8_data[0] = I2C_RdData(I2C0);
I2C_ClrIsrFlag(I2C0);
/* Wait on action complete, Read data_1, Clear Irq. flag */
while(I2C_GetIsrFlag(I2C0) == 0);
ui8_data[1] = I2C_RdData(I2C0);
I2C_ClrIsrFlag(I2C0);
/* Wait on action complete, Read data_2, Clear Irq. flag */
while(I2C_GetIsrFlag(I2C0) == 0);
ui8_data[2] = I2C_RdData(I2C0);
I2C_ClrIsrFlag(I2C0);
/* Set to NACK Mode */
I2C_SetNackMode(I2C0);
/* Wait on action complete, Read data, Clear Irq. flag */
while(I2C_GetIsrFlag(I2C0) == 0);
ui8_data[3] = I2C_RdData(I2C0);
I2C_ClrIsrFlag(I2C0);
/* Generate START on I2C bus */
I2C_StopSignal(I2C0);
/* main loop */
while (1)
{
__asm volatile ("nop");
}
}
