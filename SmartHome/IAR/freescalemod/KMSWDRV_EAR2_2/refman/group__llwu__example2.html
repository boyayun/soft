LLWU Recovery from VLLS0 Mode
LLWU Examples

This experiment demonstrates current profile of the device and timing required for recovery from the VLLS0 mode. Device exits VLLS0 mode when PTD0 is pulled low. When running this example on the TWR-KM34Z50M board then press SW1 button to resume device operation.

Measurement:
Note
In order to download new software example enter and hold device in a reset state by asserting RESET_B pin followed by power cycling and initiating downloading new software example. When downloading is about to start de-assert RESET_B pin.

#include "drivers.h"
#define LED1 PIN1 /* PTD1 */
/******************************************************************************/
void main (void)
{
SIM_Init (SIM_MODULE_LOW_POWER_PORTD_ENABLED); /* state 1 */
PMC_Init (PMC_MODULE_LVDRE_OFF_LVDINT_OFF_LVWINT_OFF_CONFIG(PMC_LVDL,PMC_LVW1),
PMC_INTREG_BGEN_OFF_BGBE_OFF_CONFIG, PRI_LVL0, (PMC_CALLBACK)NULL);
SMC_Init (SMC_MODULE_VLPS_ON_VLLS_ON_CONFIG);
PORT_Init (PORTD, PORT_MODULE_BUTTON_IRQ_MODE, PIN0);
GPIO_Init (GPIOD, GPIO_INP_MODE , PIN0);
if (PMC_GetACKISO()) { PMC_SetACKISO(); } /* release I/O states */
/* transition to VLPR mode */
arch_delay (5000l);
SMC_SetMode(VLPR); /* state 2 */
arch_delay (5000l);
/* initialize LLWU to PTD0 falling edge active */
LLWU_Init(LLWU_MODULE_PTD0_FALLING_EDGE_ENABLED, PRI_LVL0, (LLWU_CALLBACK)NULL);
EnableInterrupts();
/* main programme loop */
while (1)
{
SMC_SetMode(VLLS0_POR_OFF); /* state 3 */
}
}
