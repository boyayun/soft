RCM Example
RCM Driver

This example demonstrates typical use of the Reset Control Module (RCM) driver.

Source code:
/******************************************************************************
* (c) Copyright 2010-2013, Freescale Semiconductor Inc.
* ALL RIGHTS RESERVED.
******************************************************************************
* rcm_test.c
******************************************************************************/
#include "drivers.h"
#pragma diag_suppress=Pa082
void main (void)
{
register uint16 tmp16;
/* reset pin filtered in all modes by a filter clocked by LPO */
RCM_Init (RCM_MODULE_FILT_CLOCKED_BY_LPO_EN_IN_ALL_MODES_CONFIG);
/* read sources of the most recent reset */
tmp16 = RCM_GetResetSrc();
if (tmp16 & RCM_SACKERR) { __asm volatile ("nop"); }
if (tmp16 & RCM_MDM_AP ) { __asm volatile ("nop"); }
if (tmp16 & RCM_SW ) { __asm volatile ("nop"); }
if (tmp16 & RCM_LOCKUP ) { __asm volatile ("nop"); }
if (tmp16 & RCM_POR ) { __asm volatile ("nop"); }
if (tmp16 & RCM_PIN ) { __asm volatile ("nop"); }
if (tmp16 & RCM_LOL ) { __asm volatile ("nop"); }
if (tmp16 & RCM_LOC ) { __asm volatile ("nop"); }
if (tmp16 & RCM_LVD ) { __asm volatile ("nop"); }
if (tmp16 & RCM_WAKEUP ) { __asm volatile ("nop"); }
while(1);
}
