SWISR Example
SWISR Driver

This example demonstrates typical use of the Software Interrupt (SWISR) Driver.

Source code:
/******************************************************************************
* (c) Copyright 2010-2013, Freescale Semiconductor Inc.
* ALL RIGHTS RESERVED.
******************************************************************************
* swisr_test.c
******************************************************************************/
#include "drivers.h"
static void sw0_callback (void);
static void sw1_callback (void);
/******************************************************************************/
void main (void)
{
/* install software interrupt handler(s) */
SWISR_HandlerInit (0, PRI_LVL2, sw0_callback);
SWISR_HandlerInit (1, PRI_LVL2, sw1_callback);
/* enable interrupts on global level */
EnableInterrupts();
/* call first handler */
SWISR_HandlerCall (0,0);
while(1);
}
/******************************************************************************
* Software ISR callback *
******************************************************************************/
static void sw0_callback (void)
{
SWISR_HandlerCall (1,0);
}
static void sw1_callback (void)
{
SWISR_HandlerCall (0,0);
}
/******************************************************************************
* End of module *
******************************************************************************/
