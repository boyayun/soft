LLWU Pin Falling Edge Wakeup
LLWU Examples

This example demonstrates typical use of the Low-Leakage Wakeup Unit (LLWU) to exit selected VLLSx mode via reset condition. The software example allows selection of the STOP, VLPS, VLLS3, VLLS2, VLLS1 or VLLS0 mode for device wakeup. The power consumption has been measured on a special validation board populated with MKM34Z128CLL5 (Rev 2.0) device:

Power Down Mode Idd+Idda+(Ibat=1.9uA)
STOP 311.1uA
Very Low Power Stop (VLPS) 6.8uA
Very Low Leakage Stop 3 (VLLS3) 3.8uA
Very Low Leakage Stop 2 (VLLS2) 3.7uA
Very Low Leakage Stop 1 (VLLS1) 3.2uA
Very Low Leakage Stop 0 (VLLS0-POR_ON) 2.6uA
Very Low Leakage Stop 0 (VLLS0-POR_OFF)2.3uA

Idd+Idda+Ibat measured on a special validation board using METEX M-4650CR Multimeter.*

#include "drivers.h"
/* Button definitions */
#define BTN PIN0 /* PTD0 */
#define LED PIN1 /* PTD1 */
/* Callback function declaration */
static void port_callback (PORT_CALLBACK_SRC src, uint8 pin);
/******************************************************************************/
void main (void)
{
SIM_Init (SIM_MODULE_LOW_POWER_PORTD_ENABLED);
PMC_Init (PMC_MODULE_LVDRE_OFF_LVDINT_OFF_LVWINT_OFF_CONFIG(PMC_LVDL,PMC_LVW1),
PMC_INTREG_BGEN_OFF_BGBE_OFF_CONFIG, PRI_LVL0, (PMC_CALLBACK)NULL);
SMC_Init (SMC_MODULE_VLPS_ON_VLLS_ON_CONFIG);
IRTC_Init(IRTC_MODULE_COMP_OFF_CONFIG(LFSR_CLK_2HZ, 4),
IRTC_TAMPER_PIN_PASSIVE_POL_LOW_CONFIG(TAMPER_FILT_CLK_2HZ,4),
IRTC_TAMPER_PIN_PASSIVE_POL_LOW_CONFIG(TAMPER_FILT_CLK_2HZ,4),
IRTC_TAMPER_PIN_PASSIVE_POL_LOW_CONFIG(TAMPER_FILT_CLK_2HZ,4),
0);
PORT_Init (PORTD, PORT_MODULE_BUTTON_IRQ_MODE, BTN);
GPIO_Init (GPIOD, GPIO_INP_MODE , BTN);
PORT_Init (PORTD, PORT_MODULE_LED_MODE, LED);
GPIO_Init (GPIOD, GPIO_OUT_LOGIC1_MODE, LED);
if (PMC_GetACKISO()) { PMC_SetACKISO (); } /* release I/O states */
/* toggle LED to notify reset condition */
GPIO_Clr (GPIOD, LED);
arch_delay (5000000l);
GPIO_Set (GPIOD, LED);
/* install PORTx interrupt */
PORT_InstallCallback (PRI_LVL3, port_callback);
LLWU_Init(LLWU_MODULE_PTD0_FALLING_EDGE_ENABLED, PRI_LVL0, (LLWU_CALLBACK)NULL);
EnableInterrupts();
/* main programme loop */
while (1)
{
// SMC_SetMode(STOP); /* IDD=311.1uA, IBAT=1.9uA */
// SMC_SetMode(VLPS_PSTOP1); /* IDD= 5.0uA, IBAT=1.9uA */
// SMC_SetMode(VLPS_PSTOP2); /* IDD= 5.0uA, IBAT=1.9uA */
SMC_SetMode(VLPS); /* IDD= 4.9uA, IBAT=1.9uA */
// SMC_SetMode(VLLS3); /* IDD= 1.9uA, IBAT=1.9uA */
// SMC_SetMode(VLLS2); /* IDD= 1.8uA, IBAT=1.9uA */
// SMC_SetMode(VLLS1); /* IDD= 1.3uA, IBAT=1.9uA */
// SMC_SetMode(VLLS0_POR_ON ); /* IDD= 0.7uA, IBAT=1.9uA */
// SMC_SetMode(VLLS0_POR_OFF); /* IDD= 0.4uA, IBAT=1.9uA */
}
}
/* callback functions */
void port_callback (PORT_CALLBACK_SRC src, uint8 pin)
{
if (src == PORTD_CALLBACK)
{
if (pin == BTN)
{
GPIO_Clr (GPIOD, LED);
arch_delay (1000000l);
GPIO_Set (GPIOD, LED);
}
}
}
